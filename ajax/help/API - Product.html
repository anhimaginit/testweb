addProduct(
 
{product note => PRODUCT NOTE, product tags => [{tag 1, tag 2,…}] , product taxable => PRODUCT_TAXABLE , product updated by => PRODUCT_UPDATED_BY , prod class => PROD_CLASS , prod cost => PROD_COST, prod desc short  => PROD_DESC_SHORT , prod height  => PROD_HEIGHT ,  prod inactive  => (true| false) , prod length => PROD_LENGTH , prod name => PROD_NAME , prod price  => PROD_PRICE , prod type  =>  PROD_TYPE , prod visible => PROD_VISIBLE , prod weight => PROD_WEIGHT, prod width =>  PROD_WIDTH , sku  => SKU , photopath => PHOTOPATH }
